void cmatrix(int c[3][3]) {
    for (i = 0; i < 3; i++) {

            printf("%d\t", c[i][j]);
        }
        printf("\n");
    }
}


//<<<<<<<<<<<<uppertriangularmatrix>>>>>>>>>>>
void uptri(int matrix[3][3]) {
    printf("Upper Triangular Matrix:\n");
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            if (i <= j) {
                printf("%d\t", matrix[i][j]);
            } else {
                printf("0\t");
            }
        }
        printf("\n");
    }
}


//<<<<lower triangular matrix>>>>>>>>>>>>>>>>
void lowtri(int matrix[3][3]) {
    printf("Lower Triangular Matrix:\n");
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            if (i >= j) {
                printf("%d\t", matrix[i][j]);
            } else {
                printf("0\t");
            }
        }
        printf("\n");
    }
}


//<<<<<<<<<transpose of matrix>>>>>>>>>>>>
void transpose(int matrix[3][3]) {
    printf("Transpose of Matrix:\n");
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            printf("%d\t", matrix[j][i]);
        }
        printf("\n");
    }
}


//<<<<<<<<<check whether matrix is symmetric or not>>>>>>>>
int isSymmetric(int matrix[3][3]) {
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            if (matrix[i][j] != matrix[j][i]) {
                return 0;
            }
        }
    }
    return 1;
}

//<<<<<<<<sum of diagonal>>>>>>>>>>>>>>>>>>>>>>>>

int sumDiagonal(int matrix[3][3]) {
    int sum = 0;
    for (i = 0; i < 3; i++) {
        sum += matrix[i][i];
    }
    return sum;
}

void addmatrix(int a[3][3], int b[3][3], int c[3][3]) {
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            c[i][j] = a[i][j] + b[i][j];
        }
    }
    printf("Sum of matrices:\n");
    cmatrix(c);
}

void subsmatrix(int a[3][3], int b[3][3], int c[3][3]) {
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            c[i][j] = a[i][j] - b[i][j];
        }
    }
    printf("Difference of matrices:\n");
    cmatrix(c);
}

void mulmatrix(int a[3][3], int b[3][3], int c[3][3]) {
    for (i = 0; i < 3; i++) {
        for (j = 0; j < 3; j++) {
            c[i][j] = 0;
            for (int k = 0; k < 3; k++) {
                c[i][j] += a[i][k] * b[k][j];
            }
        }
    }
    printf("Product of matrices:\n");
    cmatrix(c);
}
